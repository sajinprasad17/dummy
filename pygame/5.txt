    @Test
    public void testcase2() throws IOException{
        test=extent.createTest("click on France");
        try{
            //driver.navigate().refresh();
          driver.navigate().to("https://www.expedia.com/newsroom/");
           Actions act=new Actions(driver);
           WebElement us=driver.findElement(By.xpath("//a[text()=\"United States\"]"));
           act.moveToElement(us).perform();
           test.log(Status.PASS, "Hover on united states");
           LoggerHandler.info("Hover on united states");
           driver.findElement(By.xpath("//a[text()=\"France\"]")).click();

           LoggerHandler.info("click on France");
           String expect=driver.findElement(By.xpath("(//h5[text()='Newsroom'])[1]")).getText();
           String actual="Newsroom";
           Assert.assertEquals(expect, actual);
           test.log(Status.PASS, "Validation Passed: Expected = " + actual);
           Screenshot.captureScreenShot("expedia_screenshot");
           Reporter.attachScreenshotToReport("expedia_screenshot.png", test, "France Screenshot");
           test.log(Status.PASS, "click on France");           
           LoggerHandler.info("screenshot is saved");
           
        }
        catch(Exception e){
            System.out.println(e);
           }
    }
